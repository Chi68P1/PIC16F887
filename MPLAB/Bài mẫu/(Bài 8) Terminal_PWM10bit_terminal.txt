void load_pwm_10bit(unsigned int VAL)
{
	// 10 bit
	CCPR1L = VAL/4;  // 
	DC1B1 = 0;
	DC1B0 = 0;
	if ((VAL & 1)==1) DC1B0=1;
	if ((VAL & 2)==2) DC1B1=1;
	//8 bit 
	//CCPR1L = VAL;
	TMR2ON=1;
}

{

	}
}
